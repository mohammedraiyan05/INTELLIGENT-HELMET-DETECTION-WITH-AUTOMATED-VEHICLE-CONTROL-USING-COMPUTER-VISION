#include <ESP8266WiFi.h>
#include <ESP8266WebServer.h>

/* -------- WiFi Credentials -------- */
const char* ssid = "moto g64 5G_3435";
const char* password = "0123456789";

/* -------- Web Server -------- */
ESP8266WebServer server(80);

/* -------- Pin Configuration -------- */
#define RELAY_PIN D1
#define RED_LED   D2
#define GREEN_LED D3
#define BUZZER    D4

/* -------- Function to Handle Access -------- */
void handleAccess() {

  if (!server.hasArg("status")) {
    server.send(400, "text/plain", "Missing status parameter");
    return;
  }

  String status = server.arg("status");

  if (status == "1") {
    // ✅ HELMET DETECTED → FULL ACCESS
    digitalWrite(RELAY_PIN, HIGH);
    digitalWrite(GREEN_LED, HIGH);
    digitalWrite(RED_LED, LOW);
    digitalWrite(BUZZER, LOW);

    Serial.println("ACCESS GRANTED - HELMET DETECTED");
    server.send(200, "text/plain", "ACCESS GRANTED");
  }
  else if (status == "0") {
    // ❌ HELMET NOT DETECTED → RESTRICTED ACCESS
    digitalWrite(RELAY_PIN, LOW);
    digitalWrite(GREEN_LED, LOW);
    digitalWrite(RED_LED, HIGH);
    digitalWrite(BUZZER, HIGH);

    Serial.println("ACCESS DENIED - NO HELMET");
    server.send(200, "text/plain", "ACCESS DENIED");
  }
  else {
    server.send(400, "text/plain", "Invalid status value");
  }
}

/* -------- Setup Function -------- */
void setup() {

  Serial.begin(115200);

  pinMode(RELAY_PIN, OUTPUT);
  pinMode(RED_LED, OUTPUT);
  pinMode(GREEN_LED, OUTPUT);
  pinMode(BUZZER, OUTPUT);

  // Default State
  digitalWrite(RELAY_PIN, LOW);
  digitalWrite(RED_LED, LOW);
  digitalWrite(GREEN_LED, LOW);
  digitalWrite(BUZZER, LOW);

  Serial.println("Connecting to WiFi...");

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi Connected!");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());

  // Route setup
  server.on("/access", handleAccess);

  server.begin();
  Serial.println("Server Started!");
}

/* -------- Loop Function -------- */
void loop() {
  server.handleClient();
}
